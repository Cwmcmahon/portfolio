Games

This interactive table uses data from the 2021 Week 12 Tidy Tuesday event.

Peek behind the curtain! 
games <- read_csv('https://raw.githubusercontent.com/rfordatascience/tidytuesday/master/data/2021/2021-03-16/games.csv') %>%
  select(-avg_peak_perc) %>%
  mutate(year = factor(year), month = factor(month, levels = month.name)) %>%
  group_by(year) %>%
  top_n(1000)

games %>% 
  select(`Game` = gamename, `Year` = year, `Month` = month, `Average Player Count` = avg, `Peak Player Count` = peak) %>%
  datatable(filter = "bottom", extensions = 'Buttons', options = list(
    dom = 'Bfrtip',
    pageLength = 10,
    lengthMenu = c(10, 20, 50),
    buttons = c('copy', 'csv', 'excel'),
    order = (list(list(2, 'desc'), list(3, 'desc'), list(4, 'desc'),  list(1, 'asc')))
  ))
